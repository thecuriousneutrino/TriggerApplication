175 void ReconRandomiser::CreateVertex(double * pos)
176 {
177  double x, y, r, z, xdir, ydir;
178  const int maxcount = 100;
179 
180  //create a flat distribution in r and phi
181  if(fFlatR
182  && (fXDistribution == kUniform)
183  && (fYDistribution == kUniform)) {
184  r = fRand->Uniform(0, +fMaxRPos);
185  fRand->Circle(xdir, ydir, 1);
186  x = r * xdir;
187  y = r * ydir;
188  }
189  //create a flat distribution in x and y
190  else {
191  r = fMaxRPos + 1;
192  while(r > fMaxRPos) {
193  //x
194  x = GetRandomNumber(fXDistribution, fMaxRPos, fXMean, fXWidth, maxcount);
195  //y
196  y = GetRandomNumber(fYDistribution, fMaxRPos, fYMean, fYWidth, maxcount);
197  //r
198  r = TMath::Sqrt(TMath::Power(x, 2) + TMath::Power(y, 2));
199  }
200  }
201  //z
202  z = GetRandomNumber(fZDistribution, fMaxZPos, fZMean, fZWidth, maxcount);
203 
204  pos[0] = x;
205  pos[1] = y;
206  pos[2] = z;

209 double ReconRandomiser::GetRandomNumber(Distribution_t dist, double max, double mean, double width, const int maxcount)
210 {
211  switch(dist) {
212  case (kUniform):
213  return fRand->Uniform(-max, +max);
214  break;
215  case(kGauss): {
216  double x = max + 1;
217  int count = 0; //don't get stuck in the loop forever

221  }

225  StreamToLog(WARN);

227  return +max;
228  else
229  return -max;
230  }
231  else

233  break;
234  }
235  case(kFixed):
236  return mean;
237  break;
238  default:

240  StreamToLog(WARN);
